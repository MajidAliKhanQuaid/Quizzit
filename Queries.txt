CREATE TABLE [Question] ([ID] INT NOT NULL IDENTITY(1,1) PRIMARY KEY, [QuestionText] NVARCHAR(MAX) NOT NULL, [QuestionType] INT NOT NULL, [NextQuestionID] INT)

CREATE TABLE [QuestionAnswer]([ID] INT NOT NULL IDENTITY(1,1) PRIMARY KEY, [QuestionID] INT NOT NULL, [AnswerText] NVARCHAR(MAX) NOT NULL, [NextQuestionID] INT)

CREATE TABLE [QuestionAndAnswer]([ID] INT NOT NULL IDENTITY(1,1) PRIMARY KEY, [UserID] INT NOT NULL, [QuestionAndAnswers] NVARCHAR(MAX))
